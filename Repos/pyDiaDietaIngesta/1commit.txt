package Repos.pyDiaDietaIngesta;
import java.util.Scanner;

public class pyDiaDietaIngesta {
    static Scanner scanner = new Scanner(System.in);

    public static void main(String[] args) {
        String name = askName();
        System.out.println("Patient's name: " + name);

        int day = askDay();
        System.out.println("Selected day: " + day);
    }

    public static String askName() {
        System.out.print("Enter your name: ");
        return scanner.nextLine();
    }

    public static int askDay() {
        System.out.print("Enter a day number (1-5): ");
        int day = scanner.nextInt();
        scanner.nextLine(); 

        while (day < 1 || day > 5) {
            System.out.println("Invalid day number. Please enter a number between 1 and 5.");
            System.out.print("Enter a day number (1-5): ");
            day = scanner.nextInt();
            scanner.nextLine();
        }

        return day;
    }
}
