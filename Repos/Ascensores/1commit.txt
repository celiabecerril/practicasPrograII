Creo matriz 7 filas con 6 columnas. 

package Repos.Ascensores;
public class pyAscensores {
    public static void main(String[] args) {
        // Define the number of rows and columns
        int numRows = 7;
        int numCols = 6;

        // Create a 2D array to represent the matrix
        int[][] matrix = new int[numRows][numCols];

        // Initialize the matrix
        initializeMatrix(matrix, numRows, numCols);

        // Define the floor names
        String[] floorNames = {"Planta 3", "Planta 2", "Planta 1", "Planta B", "Planta -1", "Planta -2", "Planta -3"};

        // Print the matrix
        printMatrix(matrix, numRows, numCols, floorNames);
    }

    public static void initializeMatrix(int[][] matrix, int numRows, int numCols) {
        for (int i = 0; i < numRows; i++) {
            for (int j = 0; j < numCols; j++) {
                matrix[i][j] = 0;
            }
        }
    }

    public static void printMatrix(int[][] matrix, int numRows, int numCols, String[] floorNames) {
        for (int i = 0; i < numRows; i++) {
            System.out.printf("%-10s", floorNames[i]);
            for (int j = 0; j < numCols; j++) {
                System.out.printf("%-3d", matrix[i][j]);
            }
            System.out.println();
        }
    }
}
